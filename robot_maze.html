import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    # 
    rnd = random.randrange(0, 100, 1)
    if rnd < (epsilon * 100):
        # 
        action = actions[random.randrange(-1, 3, 1)]
        
    else: 
        #  Q 
        max_val = 0
        for key in qline:
            if max_val < qline[key]:
                max_val = qline[key]
                action = key
    
    return(action)
